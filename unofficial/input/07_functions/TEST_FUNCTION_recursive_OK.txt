// Tests FUNC-006 - Function can call itself recursively
// Valid recursive function

int factorial(int n) {
    if (n < 2) {
        return 1;
    }
    return n * factorial(n - 1);
}

int fibonacci(int n) {
    if (n < 2) {
        return n;
    }
    return fibonacci(n - 1) + fibonacci(n - 2);
}

void main() {
    int fact := factorial(5);
    int fib := fibonacci(7);
    PrintInt(fact);
    PrintInt(fib);
}

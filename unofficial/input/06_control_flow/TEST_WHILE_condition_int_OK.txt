// Tests FUNC-001 - While condition must be int type
// Valid while statement with integer condition

void main() {
    int counter := 0;
    int limit := 10;

    while (counter < limit) {
        PrintInt(counter);
        counter := counter + 1;
    }

    while (1) {
        counter := counter - 1;
        if (counter < 0) {
            counter := 0;
        }
    }
}

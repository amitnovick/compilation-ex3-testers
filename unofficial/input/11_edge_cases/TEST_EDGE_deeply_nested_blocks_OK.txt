// Edge case - Deeply nested scope blocks
// Valid - Multiple levels of nested blocks

void process() {
    int a := 1;

    if (a > 0) {
        int b := 2;

        if (b > 0) {
            int c := 3;

            if (c > 0) {
                int d := 4;

                if (d > 0) {
                    int e := 5;
                    PrintInt(e);
                }
            }
        }
    }
}

void main() {
    process();
}

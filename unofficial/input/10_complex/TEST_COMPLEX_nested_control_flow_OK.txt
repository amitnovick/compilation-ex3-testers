// Complex integration test - Deeply nested control structures
// Tests nested if and while and scope handling

int factorial(int n) {
    int result := 1;
    int counter := 1;

    while (counter < n + 1) {
        result := result * counter;
        counter := counter + 1;
    }

    return result;
}

int processNumber(int num) {
    int result := 0;

    if (num > 0) {
        if (num < 10) {
            result := num * 2;
        } else {
            if (num < 20) {
                result := num * 3;
            } else {
                result := num * 4;
            }
        }
    } else {
        result := 0;
    }

    return result;
}

void main() {
    int fact := factorial(5);
    int processed1 := processNumber(5);
    int processed2 := processNumber(15);
    int processed3 := processNumber(25);

    PrintInt(fact);
    PrintInt(processed1);
    PrintInt(processed2);
    PrintInt(processed3);
}

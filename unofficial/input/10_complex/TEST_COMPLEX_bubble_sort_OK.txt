// Complex integration test - Bubble sort algorithm
// Tests arrays and loops and conditionals and operators

array IntArray = int[];

void bubbleSort(IntArray arr, int size) {
    int i := 0;
    int j := 0;
    int temp := 0;

    while (i < size) {
        j := 0;
        while (j < size - 1) {
            if (arr[j] > arr[j + 1]) {
                temp := arr[j];
                arr[j] := arr[j + 1];
                arr[j + 1] := temp;
            }
            j := j + 1;
        }
        i := i + 1;
    }
}

void main() {
    IntArray numbers := new int[5];
    numbers[0] := 64;
    numbers[1] := 34;
    numbers[2] := 25;
    numbers[3] := 12;
    numbers[4] := 22;

    bubbleSort(numbers, 5);
    PrintInt(numbers[0]);
    PrintInt(numbers[4]);
}
